```csharp
// ViewModel binding pattern
public class DashboardViewModel : INotifyPropertyChanged
{
    private DashboardManager _manager;
    
    public ObservableCollection<SystemMetric> Metrics { get; set; }
    
    private async Task UpdateMetrics()
    {
        var snapshot = _manager.UpdateMetrics();
        Metrics.Clear();
        foreach (var metric in snapshot.SystemMetrics)
        {
            Metrics.Add(metric);
        }
    }
}
```
