```csharp
public void SpawnKanji(string character, Point position)
{
    var kanji = new TextBlock { Text = character };
    
    // Opacity fade
    var fadeAnim = new DoubleAnimation(1, 0, TimeSpan.FromSeconds(2));
    
    // Y position up
    var slideAnim = new ThicknessAnimation(
        new Thickness(position.X, position.Y, 0, 0),
        new Thickness(position.X, position.Y - 100, 0, 0),
        TimeSpan.FromSeconds(2));
    
    // Rotation
    var rotateAnim = new DoubleAnimation(0, 360, TimeSpan.FromSeconds(2));
    
    kanji.BeginAnimation(OpacityProperty, fadeAnim);
    kanji.BeginAnimation(FrameworkElement.MarginProperty, slideAnim);
    kanji.BeginAnimation(RotateTransform.AngleProperty, rotateAnim);
}
```

**Code Pattern:**
```csharp
public class MonadoBlade : UserControl
{
    private const string BladeShapePath = "M 0,0 L 100,0 L 50,-200 Z"; // Triangle blade
    
    public void PlayGrowAnimation()
    {
        var scaleAnim = new DoubleAnimation(1.0, 1.2, TimeSpan.FromMilliseconds(400))
        {
            EasingFunction = new CubicEase { EasingMode = EasingMode.EaseOut }
        };
        
        var glowAnim = new DoubleAnimation(0.5, 1.0, TimeSpan.FromMilliseconds(400))
        {
            EasingFunction = new CubicEase { EasingMode = EasingMode.EaseOut }
        };
        
        _bladeTransform.BeginAnimation(ScaleTransform.ScaleXProperty, scaleAnim);
        _bladeTransform.BeginAnimation(ScaleTransform.ScaleYProperty, scaleAnim);
        _glowBrush.BeginAnimation(OpacityProperty, glowAnim);
    }
    
    public void PlayRippleEffect(Point clickPoint)
    {
        var ripple = new Ellipse
        {
            Width = 10, Height = 10,
            Fill = new SolidColorBrush(Colors.Cyan)
        };
        
        Canvas.SetLeft(ripple, clickPoint.X);
        Canvas.SetTop(ripple, clickPoint.Y);
        
        var scaleAnim = new DoubleAnimation(1.0, 5.0, TimeSpan.FromMilliseconds(500));
        var opacityAnim = new DoubleAnimation(1.0, 0.0, TimeSpan.FromMilliseconds(500));
        
        ripple.RenderTransform = new ScaleTransform();
        ripple.BeginAnimation(ScaleTransformProperty, scaleAnim);
        ripple.BeginAnimation(OpacityProperty, opacityAnim);
    }
}
```

```csharp
        AddProvider("Claude", "anthropic", ProviderType.BuiltIn);
        AddProvider("GPT-4", "openai", ProviderType.BuiltIn);
        AddProvider("Hermes", "local", ProviderType.BuiltIn);
        AddProvider("Local LLM", "hyper-v", ProviderType.LocalLLM);
        
        // Cloud providers (dynamic registration)
        LoadCloudProviders();
        
        // WSL2 hub
        AddProvider("WSL2 Hub", "wsl2", ProviderType.LocalLLM);
    }
    
    private void AddProvider(string name, string type, ProviderType category)
    {
        var card = new ProviderCard
        {
            Title = name,
            Status = GetProviderStatus(type),
            ResponseTime = GetAverageResponseTime(type),
            Icon = GetProviderIcon(type)
        };
        
        _providers[name] = card;
        ProviderGrid.Children.Add(card);
    }
}
```
