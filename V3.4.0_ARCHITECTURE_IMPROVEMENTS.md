## 1. SERVICE SCALING PATTERNS

### 1.1 Vertical vs. Horizontal Scaling

**v3.3.0 State** (Single-node):
- Linear throughput improvement: +20-50%
- Max capacity: 1 server × optimized
- Bottleneck: Single CPU thread pool

**v3.4.0 Target** (Multi-node):
- Horizontal scaling: N × servers
- Linear capacity increase with node count
- Distributed coordination via event bus

### 1.3 Scaling Patterns

#### Pattern 1: Cache-Aside (Lazy Loading)

```csharp
public sealed class DistributedCache
{
    private readonly IConnectionMultiplexer _redis;
    private readonly IDatabase _db;
    
    public async Task<T> GetOrComputeAsync<T>(
        string key, 
        Func<Task<T>> factory,
        TimeSpan expiration = default)
    {
        // 1. Try to get from distributed cache
        var cached = await _db.StringGetAsync(key);
        if (cached.HasValue)
        {
            return JsonConvert.DeserializeObject<T>(cached.ToString());
        }
        
        // 2. Cache miss: compute value
        var value = await factory();
        
        // 3. Store in cache
        var json = JsonConvert.SerializeObject(value);
        await _db.StringSetAsync(key, json, expiration);
        
        return value;
    }
}

// Usage:
var profile = await _cache.GetOrComputeAsync(
    $"profile:{profileId}",
    () => _profileRepository.GetByIdAsync(profileId),
    TimeSpan.FromHours(1)
);
```

#### Pattern 2: Write-Through Cache

```csharp
public sealed class WriteThrough<T>
{
    private readonly IDistributedCache _cache;
    private readonly IRepository<T> _repository;
    
    public async Task SaveAsync(string key, T entity)
    {
        // 1. Write to database first
        await _repository.SaveAsync(entity);
        
        // 2. Update cache
        var json = JsonConvert.SerializeObject(entity);
        await _cache.SetAsync(key, Encoding.UTF8.GetBytes(json));
        
        // 3. Publish invalidation event
        await _eventBus.PublishAsync(new CacheInvalidationEvent 
        { 
            Key = key,
            Timestamp = DateTime.UtcNow 
        });
    }
}
```

#### Pattern 3: Cache Coherency

```csharp
public sealed class CacheCoherencyManager
{
    private readonly IDistributedCache _cache;
    private readonly IEventBus _eventBus;
    
    public async Task InvalidateAsync(string key)
    {
        // 1. Remove from local cache
        await _cache.RemoveAsync(key);
        
        // 2. Publish invalidation to all nodes
        await _eventBus.PublishAsync(new CacheInvalidationEvent 
        { 
            Key = key,
            NodeId = Environment.MachineName,
            Timestamp = DateTime.UtcNow 
        });
    }
    
    public async Task OnCacheInvalidationEventAsync(CacheInvalidationEvent evt)
    {
        // If from different node, apply locally
        if (evt.NodeId != Environment.MachineName)
        {
            await _cache.RemoveAsync(evt.Key);
        }
    }
}
```

### 2.2 Cache Invalidation Protocol

**Time-based Expiration** (simple):
```csharp
// All cache entries expire after 1 hour
await _cache.SetAsync(key, value, TimeSpan.FromHours(1));
```

**Event-based Invalidation** (precise):
```csharp
// Invalidate on write
public async Task UpdateProfileAsync(Profile profile)
{
    // Update DB
    await _db.SaveAsync(profile);
    
    // Publish event → all nodes remove from cache
    await _eventBus.PublishAsync(new ProfileUpdated 
    { 
        ProfileId = profile.Id,
        Timestamp = DateTime.UtcNow
    });
}

// Listen for invalidation events
_eventBus.Subscribe<ProfileUpdated>(async evt =>
{
    // Remove from all node caches
    var key = $"profile:{evt.ProfileId}";
    await _cache.RemoveAsync(key);
});
```

**Hybrid Approach** (recommended):
```csharp
// Combine time + event-based
await _cache.SetAsync(
    key, 
    value, 
    absoluteExpiration: DateTime.UtcNow.AddHours(1),  // Fallback
    slidingExpiration: TimeSpan.FromMinutes(15)        // Refresh on access
);
```

## 3. EVENT BUS THROUGHPUT OPTIMIZATION

### 3.1 Current Architecture

**Synchronous Event Publishing**:
```
Publisher → Event Handler 1 → Handler completes
                → Event Handler 2 → Handler completes
                → Event Handler 3 → Handler completes

Problem: Slow handlers block publisher
```

**Optimized Architecture**:
```
Publisher → Lock-Free Queue → Event Processor
                                 ├─ Batch 1 (Handler 1, 2, 3)
                                 ├─ Batch 2 (Handler 4, 5)
                                 └─ Batch 3 (Handler 6)

Benefit: Publisher returns immediately, handlers run concurrently
```

### 3.2 Lock-Free Event Queue

```csharp
public sealed class LockFreeEventQueue<T>
{
    private readonly Atomics._uint64 _head = new();
    private readonly Atomics._uint64 _tail = new();
    private readonly T[] _buffer;
    private readonly int _mask;
    
    public LockFreeEventQueue(int capacity = 65536)
    {
        _buffer = new T[capacity];
        _mask = capacity - 1;
        _head.Value = 0;
        _tail.Value = 0;
    }
    
    public bool TryEnqueue(T item)
    {
        var tail = _tail.Value;
        var nextTail = (tail + 1) & _mask;
        
        if (nextTail == _head.Value)
        {
            return false; // Queue full
        }
        
        _buffer[tail & _mask] = item;
        _tail.Value = nextTail;
        return true;
    }
    
    public bool TryDequeue(out T item)
    {
        var head = _head.Value;
        
        if (head == _tail.Value)
        {
            item = default;
            return false; // Queue empty
        }
        
        item = _buffer[head & _mask];
        _head.Value = (head + 1) & _mask;
        return true;
    }
    
    public int ApproximateCount => 
        (int)((_tail.Value - _head.Value) & long.MaxValue);
}
```

### 3.3 Event Batching

```csharp
public sealed class BatchingEventProcessor<T>
{
    private readonly LockFreeEventQueue<T> _queue;
    private readonly List<T> _batch;
    private readonly int _batchSize;
    private readonly TimeSpan _batchTimeout;
    
    public BatchingEventProcessor(
        LockFreeEventQueue<T> queue,
        int batchSize = 100,
        TimeSpan? batchTimeout = null)
    {
        _queue = queue;
        _batch = new List<T>(batchSize);
        _batchSize = batchSize;
        _batchTimeout = batchTimeout ?? TimeSpan.FromMilliseconds(10);
    }
    
    public async Task ProcessAsync(
        Func<IReadOnlyList<T>, Task> handler,
        CancellationToken cancellationToken)
    {
        var deadline = DateTime.UtcNow + _batchTimeout;
        
        while (!cancellationToken.IsCancellationRequested)
        {
            // Try to fill batch
            while (_batch.Count < _batchSize && 
                   _queue.TryDequeue(out var item))
            {
                _batch.Add(item);
            }
            
            // Check timeout
            var remaining = deadline - DateTime.UtcNow;
            
            if (_batch.Count >= _batchSize || remaining <= TimeSpan.Zero)
            {
                if (_batch.Count > 0)
                {
                    await handler(_batch.AsReadOnly());
                    _batch.Clear();
                }
                
                deadline = DateTime.UtcNow + _batchTimeout;
            }
            else
            {
                // Wait a bit before retrying
                await Task.Delay(Math.Min(1, (int)remaining.TotalMilliseconds));
            }
        }
    }
}
```

### 3.4 Backpressure Handling

```csharp
public sealed class BackpressureEventQueue<T>
{
    private readonly LockFreeEventQueue<T> _queue;
    private readonly int _highWaterMark;
    private readonly int _lowWaterMark;
    private bool _backpressureActive;
    
    public BackpressureEventQueue(
        int capacity = 65536,
        double highWaterRatio = 0.9,
        double lowWaterRatio = 0.5)
    {
        _queue = new LockFreeEventQueue<T>(capacity);
        _highWaterMark = (int)(capacity * highWaterRatio);
        _lowWaterMark = (int)(capacity * lowWaterRatio);
    }
    
    public async Task<bool> EnqueueAsync(T item, TimeSpan timeout)
    {
        var deadline = DateTime.UtcNow + timeout;
        
        while (true)
        {
            if (_queue.TryEnqueue(item))
            {
                // Check if we crossed high water mark
                if (_queue.ApproximateCount >= _highWaterMark)
                {
                    if (!_backpressureActive)
                    {
                        _backpressureActive = true;
                        _logger.LogWarning("Backpressure: Queue near capacity");
                    }
                }
                
                return true;
            }
            
            // Queue full: wait or timeout
            if (DateTime.UtcNow >= deadline)
            {
                return false; // Timeout
            }
            
            await Task.Delay(10);
        }
    }
    
    public bool TryDequeue(out T item)
    {
        if (_queue.TryDequeue(out item))
        {
            // Check if we crossed back below low water mark
            if (_backpressureActive && _queue.ApproximateCount <= _lowWaterMark)
            {
                _backpressureActive = false;
                _logger.LogInformation("Backpressure relieved");
            }
            
            return true;
        }
        
        return false;
    }
}
```

## 4. PLUGIN SYSTEM EFFICIENCY

### 4.1 Plugin Discovery Optimization

**Current State** (Slow):
```csharp
// Reflection-based discovery on every startup
var pluginTypes = AppDomain.CurrentDomain.GetAssemblies()
    .SelectMany(a => a.GetTypes())
    .Where(t => typeof(IPlugin).IsAssignableFrom(t))
    .ToList();

// Problem: Scans all assemblies, JITs type information
```

**Optimized** (Fast):
```csharp
public sealed class PluginCache
{
    private const string CachePath = "plugins.cache.json";
    
    public async Task<List<PluginMetadata>> GetPluginsAsync()
    {
        // 1. Try to load from cache
        if (File.Exists(CachePath))
        {
            try
            {
                var json = await File.ReadAllTextAsync(CachePath);
                var cached = JsonConvert.DeserializeObject<List<PluginMetadata>>(json);
                
                // Verify cache is still valid
                if (await VerifyCacheAsync(cached))
                {
                    _logger.LogInformation("Loaded {Count} plugins from cache", cached.Count);
                    return cached;
                }
            }
            catch (Exception ex)
            {
                _logger.LogWarning(ex, "Failed to load plugin cache, rebuilding...");
            }
        }
        
        // 2. Discover plugins if cache invalid
        var plugins = DiscoverPlugins();
        
        // 3. Save cache for next startup
        await SaveCacheAsync(plugins);
        
        return plugins;
    }
    
    private List<PluginMetadata> DiscoverPlugins()
    {
        // Use Assembly.LoadFrom for explicit paths
        // Rather than scanning all AppDomain assemblies
        var pluginDir = Path.Combine(AppContext.BaseDirectory, "plugins");
        var results = new List<PluginMetadata>();
        
        foreach (var assemblyPath in Directory.GetFiles(pluginDir, "*.dll"))
        {
            try
            {
                var asm = Assembly.LoadFrom(assemblyPath);
                var pluginType = asm.GetTypes()
                    .FirstOrDefault(t => typeof(IPlugin).IsAssignableFrom(t));
                
                if (pluginType != null)
                {
                    results.Add(new PluginMetadata
                    {
                        Name = pluginType.Name,
                        Assembly = asm.FullName,
                        Type = pluginType.FullName
                    });
                }
            }
            catch (Exception ex)
            {
                _logger.LogWarning(ex, "Failed to load plugin: {Path}", assemblyPath);
            }
        }
        
        return results;
    }
}
```

### 4.2 Lazy Plugin Loading

```csharp
public sealed class LazyPluginFactory
{
    private readonly PluginCache _cache;
    private readonly Dictionary<string, Lazy<IPlugin>> _plugins = new();
    
    public async Task<IPlugin> GetPluginAsync(string pluginName)
    {
        if (_plugins.TryGetValue(pluginName, out var lazy))
        {
            return lazy.Value;
        }
        
        // Load plugin metadata from cache
        var metadata = await _cache.GetPluginAsync(pluginName);
        if (metadata == null)
        {
            throw new PluginNotFoundException($"Plugin not found: {pluginName}");
        }
        
        // Create lazy loader
        var lazyPlugin = new Lazy<IPlugin>(() =>
        {
            var asm = Assembly.Load(metadata.Assembly);
            var type = asm.GetType(metadata.Type);
            return (IPlugin)Activator.CreateInstance(type);
        });
        
        _plugins[pluginName] = lazyPlugin;
        return lazyPlugin.Value;
    }
    
    public bool IsLoaded(string pluginName)
    {
        return _plugins.TryGetValue(pluginName, out var lazy) && lazy.IsValueCreated;
    }
}
```

### 4.3 Plugin Dependency Resolution

```csharp
public sealed class PluginDependencyResolver
{
    private readonly Dictionary<string, PluginMetadata> _plugins = new();
    
    public IReadOnlyList<string> GetLoadOrder(string pluginName)
    {
        var order = new List<string>();
        var visited = new HashSet<string>();
        
        Traverse(pluginName, order, visited);
        return order;
    }
    
    private void Traverse(string plugin, List<string> order, HashSet<string> visited)
    {
        if (visited.Contains(plugin))
        {
            return; // Already processed
        }
        
        visited.Add(plugin);
        
        // Load dependencies first
        if (_plugins.TryGetValue(plugin, out var metadata))
        {
            foreach (var dep in metadata.Dependencies)
            {
                Traverse(dep, order, visited);
            }
        }
        
        // Then load this plugin
        order.Add(plugin);
    }
}
```

## 5. MONITORING & OBSERVABILITY

### 5.1 Performance Metrics Collection

```csharp
public sealed class PerformanceMetricsCollector
{
    private readonly Dictionary<string, PerformanceCounter> _counters = new();
    
    public void RecordOperation(
        string operationName,
        TimeSpan duration,
        bool success = true)
    {
        if (!_counters.TryGetValue(operationName, out var counter))
        {
            counter = new PerformanceCounter(operationName);
            _counters[operationName] = counter;
        }
        
        counter.Record(duration, success);
    }
    
    public OperationMetrics GetMetrics(string operationName)
    {
        return _counters.TryGetValue(operationName, out var counter)
            ? counter.GetMetrics()
            : null;
    }
}

public class PerformanceCounter
{
    private long _totalDuration;
    private long _successCount;
    private long _failureCount;
    private long _minDuration = long.MaxValue;
    private long _maxDuration = 0;
    private readonly List<long> _recentDurations = new();
    
    public void Record(TimeSpan duration, bool success)
    {
        var ms = (long)duration.TotalMilliseconds;
        
        Interlocked.Add(ref _totalDuration, ms);
        
        if (success)
            Interlocked.Increment(ref _successCount);
        else
            Interlocked.Increment(ref _failureCount);
        
        UpdateMinMax(ms);
        
        lock (_recentDurations)
        {
            _recentDurations.Add(ms);
            if (_recentDurations.Count > 1000)
                _recentDurations.RemoveAt(0);
        }
    }
    
    private void UpdateMinMax(long ms)
    {
        Interlocked.CompareExchange(ref _minDuration, ms, long.MaxValue);
        Interlocked.CompareExchange(ref _maxDuration, ms, 0);
    }
    
    public OperationMetrics GetMetrics()
    {
        var total = _successCount + _failureCount;
        
        List<long> durations;
        lock (_recentDurations)
        {
            durations = new List<long>(_recentDurations);
        }
        durations.Sort();
        
        return new OperationMetrics
        {
            Count = total,
            SuccessCount = _successCount,
            FailureCount = _failureCount,
            SuccessRate = total > 0 ? (double)_successCount / total : 0,
            MinDuration = _minDuration == long.MaxValue ? 0 : _minDuration,
            MaxDuration = _maxDuration,
            AverageDuration = total > 0 ? _totalDuration / total : 0,
            P50Duration = durations.Count > 0 ? durations[durations.Count / 2] : 0,
            P99Duration = durations.Count > 0 ? durations[(int)(durations.Count * 0.99)] : 0,
        };
    }
}
```
